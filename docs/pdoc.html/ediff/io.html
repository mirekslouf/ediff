  82    def read(XYdata, **kwargs):
  83        '''
  84        Read XY data as dataframe with several columns.
  85    
  86        Parameters
  87        ----------
  88        XYdata : str or pandas.DataFrame or numpy.ndarray
  89            
  90            * If XYdata = str,
  91              we assume that it is a filename
  92              of the file with several columns of data.
  93              The optional argument {dtype} determines how to read the data.
  94              The optional argument {dcols} determines how to name the columns.
  95            * If XYdata = pandas.Dataframe,
  96              we simply read the dataframe and return it.
  97              Optionally, we can change column names if {dcols} is given.
  98            * If XYdata = numpy.ndarray,
  99              we convert the array to DataFrame
 100              and define column names if {dcols} is given.
 101              
 102    
 103        Returns
 104        -------
 105        XYdata : pandas.Dataframe
 106            The DataFrame representing XYdata.
 107            Typically, the XYdata represent a XRD or ELD diffraction profile.        
 108        '''
 109        
 110        # (1) Check if the column names were defined.
 111        #  - the column names can be defined/changed when calling this func
 112        #  - typical column names for XRD profiles: [TwoTheta, S, q, Intensity]
 113        #  - typical column names for ELD profiles: [q, Iraw, Ibkg, I] 
 114        my_columns  = kwargs.get('columns') or None
 115        
 116        # (2) Read XYdata depending on the data type.
 117        #  - the data type can be filename/pd.DataFrame/np.ndarray
 118        if isinstance(XYdata, str):
 119            # XYdata is a str = file
 120            # ... comments may be present
 121            my_comment  = kwargs.get('comment') or '#'
 122            # ... header should be None if there are no column headers!
 123            my_header   = kwargs.get('header') or None
 124            # ... skiprows can be defined for completness and flexibility
 125            my_skiprows = kwargs.get('skiprows') or 0
 126            # ... separator can be defined, defalt is any whitespace
 127            my_sep      = kwargs.get('sep') or r'\s+'
 128            # ... usecols may be defined, default is the first two columns
 129            my_usecols  = kwargs.get('usecols') or None
 130            # (c) read the file with pd.read_csv and the **kwargs from above
 131            # (Trick: we are not sure, if the file has header
 132            # (Step1: read with header=None
 133            # (Step2: try to convert df to float
 134            # (Step3: if it fails, 1st line was header => re-read with header=0
 135            # (Note: this works also with comment='#' => 1st non-commented line
 136            # -----
 137            # Step1: reading header given as argument OR header=None (default)
 138            df = pd.read_csv(XYdata, 
 139                comment=my_comment, header=my_header, skiprows=my_skiprows, 
 140                sep=my_sep, usecols=my_usecols)
 141            # Step2: try if the first line is header or not
 142            try:
 143                df.astype(float)
 144            except ValueError:
 145                # Step3: if conversion failed, re-read with header=0
 146                df = pd.read_csv(XYdata, 
 147                    comment=my_comment, header=0, skiprows=my_skiprows, 
 148                    sep=my_sep, usecols=my_usecols)
 149            # (d) Set column names if columns was given as argument
 150            if my_columns is not None: df.columns = my_columns
 151            return(df)
 152        elif isinstance(XYdata, pd.DataFrame):
 153            # XYdata is a DataFrame => add column names if required and return.
 154            if my_columns is not None: XYdata.columns = my_columns
 155            return(XYdata)
 156        elif isinstance(XYdata, np.ndarray):
 157            # XYdata is an array => convert to Dataframe.
 158            df = pd.DataFrame(np.transpose(XYdata))
 159            if my_columns is not None: df.columns =  my_columns
 160            return(df)
 161        elif XYdata is None:
 162            # XYdata is None => create an empty dataframe with {dcols}
 163            # * XYdata = None is a rare, but real case
 164            #   It may happen if we call: ELD = ediff.io.Profile(),
 165            #   which calls ediff.io.Data1D.read internally (and XYdata=None).
 166            # * The following command simply creates
 167            #   empty DataFrame with predefined column names,
 168            #   and it works even without column names => for {dcols} = None.
 169            df = pd.DataFrame(columns=my_columns)
 170            return(df)
 171        else:
 172            raise TypeError('Unknown format of XYdata!')

 175    def show(df, X, Y, ptype='plot',
 176             Xlabel=None, Ylabel=None, Xlim=None, Ylim=None,
 177             title=None, out_file=None, out_dpi=300,
 178             **kwargs):
 179        '''
 180        Show/plot a 1D profile in a simple and stadnard way.
 181        
 182        Parameters
 183        ----------
 184        df : pandas.Dataframe
 185            A DataFrame, from which we select columns for plotting.
 186        X : str
 187            X values for plotting, column name in {df}.
 188        Y : str
 189            Y values for plotting, column name in {df}.
 190        ptype : 'plot' or 'vlines', optional, default is 'plot'
 191            Plot type. The values 'plot' and 'vlines' are suitable for
 192            the diffraction profile and individual diffractions, respectively.
 193        Xlabel : str, optional, default is None
 194            Label of the X-axis.
 195        Ylabel : str, optional, default is None
 196            Label of the Y-axis.
 197        Xlim : float ar list/tuple of two floats, optional, default is None
 198            X-range, minimum and maximum of X-axis.    
 199            If Xlim = 300, then we plot with plt.xlim(0,300).
 200            If Xlim = (50,350), the we plot with plt.xlim(50,350).
 201        Ylim : float or list/tuple of two floats, optional, default is None
 202            Y-range, minimum and maximum of Y-axis.    
 203            If Ylim = 300, then we plot with plt.ylim(0,300).
 204            If Ylim = (50,350), the we plot with plt.ylim(50,350).
 205        title : str, optional, default is None
 206            The title of the plot.
 207        out_file : str, optional, default is None
 208            Name of the output file.
 209            If the argument is not None, the plot is saved to *output_file*.
 210        out_dpi : int, optional, default is 300
 211            Resolution of the output file.
 212        **kwargs : arguments supplied to plt.plot command
 213            Any arguments that are transferred to plt.plot command.
 214            For example: color='red', linestyle=':', linewidth='1', label='ED'
 215            
 216        Returns
 217        -------
 218        None
 219            The plot is shown in the stdout
 220            and saved to {out_file} if the arqument is given.
 221        '''
 222        
 223        # (0) Check if the X,Y are existing column names.
 224        if X not in df.columns:
 225            raise ValueError(f"Column '{X}' does not exist!")
 226        if Y not in df.columns:
 227            raise ValueError(f"Column '{Y}' does not exist!")
 228        
 229        # (1) Basic plot
 230        if ptype == 'plot':
 231            plt.plot(df[X], df[Y], **kwargs)
 232        elif ptype == 'vlines':
 233            plt.vlines(df[X], 0, df[Y], **kwargs)
 234        else:
 235            raise TypeError('Unknown plot type!')
 236            
 237        # (2) Optional arguments
 238        # Plot title :: if requested
 239        if title is not None: plt.title(title)
 240        # Axes labels :: if Xlabel,Ylabel not given, use column names instead
 241        if Xlabel is None: Xlabel = X
 242        if Ylabel is None: Ylabel = Y
 243        plt.xlabel(Xlabel)
 244        plt.ylabel(Ylabel)
 245        # Axes limits :: if not defined, use defaults
 246        if Xlim is not None: 
 247            if isinstance(Xlim, (int,float)): 
 248                plt.xlim(0,Xlim)
 249            elif isinstance(Xlim, (tuple,list)): 
 250                plt.xlim(Xlim)
 251            else:
 252                raise TypeError('Wrong type of Xlim argument!')
 253        if Ylim is not None: 
 254            if isinstance(Ylim, (int,float)): 
 255                plt.ylim(0,Ylim)
 256            elif isinstance(Ylim, (tuple,list)): 
 257                plt.ylim(Ylim)
 258            else:
 259                raise TypeError('Wrong type of Ylim argument!')
 260        
 261        # (3) Additional plot parameters
 262        plt.grid()
 263        plt.tight_layout()
 264        
 265        # (4) Save the plot if requested
 266        if out_file is not None: plt.savefig(out_file, dpi=out_dpi)
 267        
 268        # (5) Show the plot
 269        plt.show()

 329    def read(image):
 330        '''
 331        Read 2D diffraction pattern (image or array) into 2D numpy array.
 332        
 333        Parameters
 334        ----------
 335        image : str or PathLike object or numpy.ndarray
 336            Name of image that should read into numpy 2D array
 337            or directly the 2D numpy array representing the diffractogram.
 338          
 339        Returns
 340        -------
 341        arr : 2D numpy array
 342            The *arr* is the input image read to an array by means of numpy.
 343        '''
 344        
 345        # Test the input type
 346        if isinstance(image, np.ndarray):
 347            # Diffractogram is an array - just assign to arr variable.
 348            arr = image
 349        else:
 350            # Diffractogram is not an array - we expect a grayscale image.
 351            # (a) Read the image
 352            # (skimage should read both 8bit and 16bit, with correct img.dtype
 353            img = ski.io.imread(image)
 354            # (b) Grayscale images can be 1-channel (G) or 3-channel (R=G=B)
 355            # (the following code converts the image to 1-channel grayscale
 356            img = img[..., 0] if img.ndim == 3 else img
 357            # (c) Just to be sure, check if the image is 8bit or 16bit
 358            # (if it is 8bit in 16bit containers, we leave it like this
 359            # (ImageJ sees it as 8bit even if dtype is uint16
 360            if img.dtype not in (np.uint8, np.uint16):
 361                raise TypeError("Only 8bit and 16bit images allowed!")
 362            # (d) Assign img to arr if everything was Ok so far
 363            arr = img
 364        
 365        # Return the diffractogram in the form of 2D array.
 366        return(arr)

 423    def save(image, out_file, itype=None, icut=None):
 424        '''
 425        Save 2D data (usually a 2D diffraction pattern) as an image file.
 426
 427        Parameters
 428        ----------
 429        image : numpy.array
 430            A numpy.array object corresponding to an image.
 431            Typically, the array represents a 2D diffraction pattern,
 432            which had been read by means by means of ediff.io funcs.
 433        out_file : str or PathLike object
 434            The name of the file into which the image/array will be saved.
 435            The image format is deduced from the extension of the {out_file}.
 436        itype : '8bit or '16bit', optional, default is '16bit'
 437            Bit depth of the saved image (8bit or 16bit grayscale).
 438            The bit depth is auto-estimated by skimage.io.save
 439            from {image}.dtype but it can be also set here explicitly.
 440            The explicit setting is maintained mostly for historical reasons.
 441        icut : int or float, optional, default is None
 442            Intensity cut.
 443            If {icut} = 200, then all intensity values >200 are set to 200.
 444
 445        Returns
 446        -------
 447        None
 448            The input image/array is saved in {out_file}.
 449        '''
 450        # Apply {icut} if requested.
 451        if icut is not None:
 452            image = np.where(image > icut, icut, image)
 453        
 454        # Check the image type and hard-convert to required type if needed
 455        if itype == '8bit':
 456            if image.dtype != np.uint8: image = ski.img_as_ubyte(image)
 457        elif itype == '16bit':
 458            if image.dtype != np.uint16: image = ski.img_as_uint(image)
 459        else:
 460            raise ValueError("Only '8bit' or '16bit' images are allowed!")
 461        
 462        # Save the image.
 463        ski.io.imsave(out_file, image)
