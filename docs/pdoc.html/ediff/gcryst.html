 67    @staticmethod
 68    def _integerize(vec, tol=1e-6):
 69        """
 70        Round a fractional vector to small integers for (hkl) or [uvw].
 71
 72        Parameters
 73        ----------
 74        vec : array-like
 75            Fractional vector
 76        tol : float
 77            Small values below tol are treated as zero
 78
 79        Returns
 80        -------
 81        tuple of ints
 82        """
 83        v = np.array(vec, dtype=float)
 84        v[np.abs(v) < tol] = 0.0
 85        if np.all(np.abs(v) < tol):
 86            return tuple(int(0) for _ in v)
 87        # Scale to smallest nonzero component = 1
 88        scale = 1.0 / np.min([abs(x) for x in v if abs(x) > tol])
 89        int_vec = tuple(int(round(x * scale)) for x in v)
 90        # Reduce by GCD of nonzero elements
 91        nz = np.array([abs(x) for x in int_vec if x != 0], dtype=int)
 92        if len(nz) > 0:
 93            g = np.gcd.reduce(nz)
 94            int_vec = tuple(int(x // g) for x in int_vec)
 95        return int_vec

168    def _convert_to_smallest_integers(uvtw):
169        # Functional programming style
170        # (why functional programming? => finding gcd of more numbers 
171        # (source/inspiration: https://stackoverflow.com/q/16628088/
172        # (1) map: for each element - multiply by 3 and round
173        uvtw = list(map(lambda x: round(x*3), uvtw))
174        # (2) reduce: for all elements - calculate greatest common divisor
175        my_gcd = reduce(gcd, uvtw)
176        # (3) map: for each element - divide by gcd and round
177        uvtw = list(map(lambda x: round(x/my_gcd) , uvtw))
178        # Return the modified list
179        return(uvtw)

221    def uvw_to_uvtw(uvw):
222        '''
223        Convert hexagonal indices: Miller [uvw] to Weber [uvtw]
224
225        Parameters
226        ----------
227        uvw : tuple or list or array of three integers
228            Miller indices [uvw]
229            defining a crystallographic direction.
230
231        Returns
232        -------
233        uvtw : tuple of four integers
234            Weber indices [uvtw] of the crystallographic direction.
235        '''
236        u,v,w = uvw
237        U = (2*u - v) / 3
238        V = (2*v - u) / 3
239        T = - (u + v) / 3
240        W = w
241        U,V,T,W = HexLattice._convert_to_smallest_integers([U,V,T,W])
242        return(U,V,T,W)

245    def uvtw_to_uvw(uvtw):
246        '''
247        Convert hexagonal indices: Weber [uvtw] to Miller [uvw].
248
249        Parameters
250        ----------
251        uvtw : tuple or list or array of four integers
252            Weber indices [uvtw]
253            defining a crystallographic direction.
254
255        Returns
256        -------
257        uvw : tuple of three integers
258            Miller indices [uvw] of the crystallographic direction.
259        '''
260        U,V,T,W = uvtw
261        u = U - T
262        v = V - T
263        w = W
264        u,v,w = HexLattice._convert_to_smallest_integers([u,v,w])
265        return(u,v,w)

273    def from_two_diffs(diff1, diff2):
274        '''
275        Calculate zone axis [uvw] from (hkl) of 2 diffractions in diff.pattern.
276
277        Parameters
278        ----------
279        diff1 : tuple or list or array of three integers
280            Miller indices (hkl) of the 1st diffraction within given zone 
281        diff2 : TYPE
282            Miller indices (hkl) of the 2nd diffraction within given zone.
283
284        Returns
285        -------
286        zone_axis : tuple of three integers
287            Miller indices [uvw] of the zone axis
288            calculated from *diff1* and *diff2*.
289        
290        Technical notes
291        ---------------
292        * Zone axis should be perpendicular
293          to any two diffractions of given zone.
294        * Therefore, it can be calculated as a vector cross-product
295          of any two diffractoins of given zone.
296        * We use np.cross to calculate the cross product
297          and convert the result to tuple for the sake of consistency
298          as all other functions in this module return vectors as tuples.
299        '''
300        zone_axis = np.cross(diff1,diff2)
301        return( tuple(zone_axis) )

304    def from_multiple_diffs(diffs):
305        '''
306        Calculate zone axis [uvw] from (hkl) of N diffractions in diff.pattern.
307
308        Parameters
309        ----------
310        diffs : list of 3-item lists = list of (hkl) of multiple diffractions
311            The list containing (hkl) indices of multiple diffractions
312            that are supposed to correspond to one zone axis [uvw].
313
314        Returns
315        -------
316        None
317            The list of vector cross-products is printed on the screen.
318            If all cross-products are the same (up to scale/sign),
319            then the zone axis [uvw] explains the presense of all *diffs*
320            in the observed diffraction pattern.
321        '''
322        for d1 in (diffs):
323            for d2  in (diffs):
324                index_d1 = diffs.index(d1) + 1
325                index_d2 = diffs.index(d2) + 1
326                if index_d1 < index_d2:
327                    print(f'd{index_d1} x d{index_d2}', end='')
328                    print(f' = {np.cross(d1,d2)}')

330    def weiss_zone_law(zone_axis):
331        '''
332        Using Wess Zone Law, calculate diffractions (hkl) for zone axis [uvw]. 
333
334        Parameters
335        ----------
336        zone_axis : tuple or list or array of three integers
337            Crystallographic direction [uvw] that defines zone axis.
338
339        Returns
340        -------
341        WZL : string
342            A condition for (hkl) indices of diffractions
343            that satisfy Weis Zone Law for given zone axis/direction [uvw].
344        
345        Technical note
346        --------------
347        * The condition is a simple string.
348        * It would be more straightforward to use a SymPy expression.
349        * Nevertheless, in this simple case the "manual symbolic calculation"
350          is quite easy and so we do not need the heavy-weight SymPy library.
351        '''
352        # WZL = Weiss Zone Law = h*u + k*v + l*w = 0
353        WZL = ''
354        u,v,w = zone_axis
355        # Process h*u part of WZL
356        if   u == 1     : WZL += 'h'
357        elif u == -1    : WZL += '-h'
358        elif abs(u) > 1 : WZL += str(u) + 'h'
359        # Process k*v part of WZL
360        if   v == 1     : WZL += ' + k'
361        elif v == -1    : WZL += ' - k'
362        elif v > 1      : WZL += ' + ' + str(v) + 'k'
363        elif v < -1     : WZL += ' - ' + str(v) + 'k'
364        # Process l*w part of WZL
365        if   w == 1     : WZL += ' + l'
366        elif w == -1    : WZL += ' - l'
367        elif w > 1      : WZL += ' + ' + str(w) + 'l'
368        elif w < -1     : WZL += ' - ' + str(w) + 'l'
369        # Add right-hand side of WZL
370        WZL += ' = 0'
371        # Return the result
372        return(WZL)

332        '''
333        Using Wess Zone Law, calculate diffractions (hkl) for zone axis [uvw]. 
334
335        Parameters
336        ----------
337        zone_axis : tuple or list or array of three integers
338            Crystallographic direction [uvw] that defines zone axis.
339
340        Returns
341        -------
342        WZL : string
343            A condition for (hkl) indices of diffractions
344            that satisfy Weis Zone Law for given zone axis/direction [uvw].
345        
346        Technical note
347        --------------
348        * The condition is a simple string.
349        * It would be more straightforward to use a SymPy expression.
350        * Nevertheless, in this simple case the "manual symbolic calculation"
351          is quite easy and so we do not need the heavy-weight SymPy library.
352        '''
353        # WZL = Weiss Zone Law = h*u + k*v + l*w = 0
354        WZL = ''
355        u,v,w = zone_axis
356        # Process h*u part of WZL
357        if   u == 1     : WZL += 'h'
358        elif u == -1    : WZL += '-h'
359        elif abs(u) > 1 : WZL += str(u) + 'h'
360        # Process k*v part of WZL
361        if   v == 1     : WZL += ' + k'
362        elif v == -1    : WZL += ' - k'
363        elif v > 1      : WZL += ' + ' + str(v) + 'k'
364        elif v < -1     : WZL += ' - ' + str(v) + 'k'
365        # Process l*w part of WZL
366        if   w == 1     : WZL += ' + l'
367        elif w == -1    : WZL += ' - l'
368        elif w > 1      : WZL += ' + ' + str(w) + 'l'
369        elif w < -1     : WZL += ' - ' + str(w) + 'l'
370        # Add right-hand side of WZL
371        WZL += ' = 0'
372        # Return the result
373        return(WZL)
