  55
  56    Parameters
  57    ----------
  58    input_image : str, path, or numpy.array
  59        The input image representing a 2D diffraction pattern, either as a 
  60        file path or a NumPy array.
  61    determination : callable, optional, default is None
  62        Method used for center determination
  63        = the initial estimate of center coordinates.
  64    refinement : callable, optional, default is None
  65        Method used for center refinement
  66        = refining the calculated center position interactively.
  67    in_file : str, optional, default is None

  70    out_file : str, optional, default is None
  71        Filename of the text file

  84        within which we will search the intensity center.
  85    cintensity : float, optional, default is 0.8

  89    messages : bool, optional, default is False

 291    def load_results(self):
 292        '''
 293        Load results from a specified text file.
 294    
 295        This method reads the coordinates from a text file defined by the 
 296        `out_file` attribute. It extracts pairs of coordinates (x1, y1) and 
 297        (x2, y2) from each line in the file. The extracted values are stored in 
 298        lists, allowing for the retrieval of multiple results.The most recent 
 299        values of x1, y1, x2, and y2 can be stored as instance variables
 300        for easy access.
 301    
 302        Parameters
 303        ----------
 304        None
 305    
 306        Returns
 307        -------
 308        None
 309    
 310        Notes
 311        -----
 312        - The results are expected to be in the format:
 313            x1: <value>, y1: <value>
 314            x2: <value>, y2: <value>
 315        - If multiple sets of coordinates are found, they are stored in lists,
 316        and only the last set can be accessed as instance variables.
 317        '''        
 318        
 319        if self.out_file is not None and os.path.isfile(self.out_file):
 320            with open(self.out_file, 'r') as f:
 321                lines = f.readlines()
 322                
 323                # Initialize lists to hold multiple values
 324                x1_values = []
 325                y1_values = []
 326                x2_values = []
 327                y2_values = []
 328    
 329                for line in lines:
 330                    # Strip and split each line to get the coordinates
 331                    parts = line.strip().split(',')
 332                    if len(parts) == 2:
 333                        # Extract x1 and y1 or x2 and y2 based on line content
 334                        if 'x1' in parts[0]:
 335                            x1 = float(parts[0].split(':')[1].strip())
 336                            y1 = float(parts[1].split(':')[1].strip())
 337                            x1_values.append(x1)
 338                            y1_values.append(y1)
 339                        elif 'x2' in parts[0]:
 340                            x2 = float(parts[0].split(':')[1].strip())
 341                            y2 = float(parts[1].split(':')[1].strip())
 342                            x2_values.append(x2)
 343                            y2_values.append(y2)
 344    
 345                # Store the last set of values as instance variables 
 346                self.x1 = x1_values
 347                self.y1 = y1_values
 348                self.x2 = x2_values
 349                self.y2 = y2_values
 350    
 351                # Store the last loaded values:
 352                if x1_values and y1_values and x2_values and y2_values:
 353                    self.x1 = x1_values[-1]
 354                    self.y1 = y1_values[-1]
 355                    self.x2 = x2_values[-1]
 356                    self.y2 = y2_values[-1]
 357    
 358        else:
 359            print("Output file does not exist or is not specified.")

 362    def get_circle_pixels(self, xc, yc, radius, num_points=360):
 363        '''         
 364        Get coordinates of pixels defining circle border
 365    
 366        Parameters
 367        ----------
 368        self.image_path : str
 369            direct path to a image with diffraction patterns
 370        xc : float64
 371            x-coordinate of the detected center
 372        yc : float64
 373            y-coordinate of the detected center
 374        radius : float64
 375            radius of the detected center
 376        num_points : float64 
 377            number of border points. The default is 360
 378        
 379        Returns
 380        -------
 381        x : array of float64
 382            x-coordinates of pixels from circle border
 383        y : array of float64
 384            y-coordinates of pixels from circle border
 385            
 386        '''
 387        
 388        # Generate angles from 0 to 2*pi
 389        theta = np.linspace(0, 2*np.pi, num=num_points)  
 390                
 391        # Calculate x,y-coordinates of points on the actual circle border
 392        x_actual = xc + radius * np.cos(theta)
 393        y_actual = yc + radius * np.sin(theta)
 394        
 395        return x_actual, y_actual

 398    def intensity_sum(self, image, px, py, pr):
 399        ''' 
 400        Summation of intensity values of pixels of a diffraction pattern.
 401
 402        Parameters
 403        ----------
 404        image : array of uint8
 405            image from which the diffraction pattern has been detected.
 406        px : float64
 407            x-coordinate of the center of the diffraction pattern.
 408        py : float64
 409            y-coordinate of the center of the diffraction pattern.
 410        pr : float64
 411            radius of the diffraction pattern.
 412
 413        Returns
 414        -------
 415        s : float64
 416            intensity sum
 417
 418        '''
 419        # Extract pixels on the circle border
 420        pxc, pyc = self.get_circle_pixels(px, py, pr)
 421        pxc = np.array(pxc, dtype=int)
 422        pyc = np.array(pyc, dtype=int)
 423        
 424        # Calculate sum using the filtered values
 425        s = np.sum(image[pyc, pxc])/len(pxc)
 426
 427        return s

 535    def preprocess(self,preInit=0,preHough=0,preManual=0,preVar=0,preSum=0,preInt=0):  
 536        """ FOR AUTOMATIC METHODS OPTIMIZATION AND MORE UNIVERSAL SOLUTIONS
 537        Function for input image preprocessing based on the methods 
 538        defined in the class initialization - self.detection_method, 
 539        self.correction_method.
 540
 541        Parameters
 542        ----------
 543        preInit : bool, optional

 546            specified, the detection and refinement will be performed on 
 547            original image. The default is 0.

 549            Perform preprocessing for automatic detection via Hough transform. 
 550

 553        -------

 558        """

 560        # Flags

 615                            sigma=0.2, 
 616                            low_threshold=2.5*np.median(self.parent.to_refine),

 618                    
 619                    # Dilate the edges to connect them
 620                    selem = sk.morphology.disk(5)

 732    def detection_intensity(self, csquare, cintensity, plot_results=0):
 733        '''
 734        Find center of intensity/mass of an array.
 735        
 736        Parameters
 737        ----------
 738        arr : 2D-numpy array
 739            The array, whose intensity center will be determined.
 740        csquare : int, optional, default is 20
 741            The size/edge of the square in the (geometrical) center.
 742            The intensity center will be searched only within the central square.
 743            Reasons: To avoid other spots/diffractions and
 744            to minimize the effect of possible intensity assymetry around center. 
 745        cintensity : float, optional, default is 0.8
 746            The intensity fraction.
 747            When searching the intensity center, we will consider only
 748            pixels with intensity > max.intensity.
 749            
 750        Returns
 751        -------
 752        xc,yc : float,float
 753            XY-coordinates of the intensity/mass center of the array.
 754            Round XY-coordinates if you use them for image/array calculations.
 755        '''  
 756        
 757        # (1) Get image/array size
 758        image = np.copy(self.parent.to_refine)
 759        
 760                
 761        arr = np.copy(image)
 762        xsize,ysize = arr.shape
 763        
 764        # (2) Calculate borders around the central square
 765        xborder = (xsize - csquare) // 2
 766        yborder = (ysize - csquare) // 2
 767        
 768        # (3) Create the central square,
 769        # from which the intensity center will be detected
 770        arr2 = arr[xborder:-xborder,yborder:-yborder].copy()
 771        
 772        # (4) In the central square,
 773        # set all values below cintenstity to zero
 774        arr2 = np.where(arr2>np.max(arr2)*cintensity, arr2, 0)
 775        
 776        # (5) Determine the intensity center from image moments
 777        # see image moments in...
 778        # skimage: https://scikit-image.org/docs/dev/api/skimage.measure.html
 779        # wikipedia: https://en.wikipedia.org/wiki/Image_moment -> Centroid
 780        # ---
 781        # (a) Calculate 1st central moments of the image
 782        M = moments(arr2,1)
 783        # (b) Calculate the intensity center = centroid according to www-help
 784        (self.x, self.y) = (M[1,0]/M[0,0], M[0,1]/M[0,0])
 785        # (c) We have centroid of the central square
 786        # but we have to recalculate it to the whole image
 787        (self.x, self.y) = (self.x + xborder, self.y + yborder)
 788        # (d) Radius of a diffraction ring is hardcoded to 100 here
 789        # TODO: consider improving/correcting/removing in the future
 790        self.r = 100
 791        
 792        # (6) User information (if required)
 793        if self.parent.messages:
 794            print("--- IntensityCenter = center of intensity/mass ---")
 795            print(f"Center coordinates [x, y]: [{self.x:.3f}, {self.y:.3f}]")
 796            print("---")
 797
 798        # (7) Plot results (if required)
 799        if plot_results == 1:
 800           self.visualize_center(self.x, self.y, self.r) 
 801                
 802        # (8) Return the results:
 803        # a) XY-coordinates of the center
 804        # b) radius of the circle/diff.ring,
 805        #    from which the center was determined
 806        # ! radius of the circle is just hardcoded here - see TODO note above
 807        return(self.x, self.y, self.r)

 810    def detection_Hough(self, plot_results=0):
 811        '''        
 812        Perform Hough transform to detect center of diffraction patterns.
 813        This is a method to automatically detect circular diffraction patterns
 814        
 815        Parameters
 816        ----------
 817        plot_results : int, binary
 818            Plot the pattern determined by pixels selected by the user.
 819            Default is 1. To cancel visualization, set plot_results = 0.

 823        self.x : float64
 824            x-coordinate of the detected center
 825        self.y : float64
 826            y-coordinate of the detected center
 827        self.r : float64
 828            radius of the detected center
 829                                    
 830        '''
 831        ## Image preprocessing
 832        im = np.copy(self.parent.to_refine)
 833        
 834        # if the brightness of the image is small enough, pixel values greater
 835        # than 50 will be set to 0 -- removal of the beam stopper influence
 836        
 837        if self.parent.heq == 0:
 838            if sum(sum(im)) < 150000:
 839                    max_indices = np.where(im > 50)
 840        
 841                    row_idx = max_indices[0]
 842                    col_idx = max_indices[1]
 843        
 844                    im[row_idx, col_idx] = 0    
 845                
 846            # Detect edges using the Canny edge detector
 847            edges = sk.feature.canny(im, 
 848                                     sigma=0.2, 
 849                                     low_threshold=80, 
 850                                     high_threshold=100)
 851        elif self.heq == 1:
 852            if sum(sum(im)) > 40000:
 853                max_indices = np.where(im > 50)
 854
 855                row_idx = max_indices[0]
 856                col_idx = max_indices[1]
 857
 858                im[row_idx, col_idx] = 0    
 859            
 860            # Detect edges using the Canny edge detector
 861            edges = sk.feature.canny(im, 
 862                                     sigma=0.2, 
 863                                     low_threshold=0.80, 
 864                                     high_threshold=1)
 865        
 866        
 867        # Define the radii range for the concentric circles
 868        # (set empirically based on the available pictures)
 869        min_radius = 40
 870        max_radius = 200
 871        radius_step = 10
 872        radii = np.arange(min_radius, max_radius + radius_step, radius_step)
 873
 874        ### Perform the Hough transform to detect circles
 875        # Circle detection involves converting edge pixels into parameter space, 
 876        # where each point represents a possible circle center and radius. 
 877        # The circles are then identified as peaks in the parameter space, 
 878        # enabling accurate detection of circular shapes in the image.
 879        hough_res = hough_circle(edges, radii)
 880
 881        # Extract the circle peaks
 882        _, self.x, self.y, self.r = hough_circle_peaks(hough_res, 
 883                                                            radii, 
 884                                                            total_num_peaks=1)
 885        
 886        
 887        # User information:
 888        if self.parent.messages:
 889            print("------------ Diffraction pattern detection via Hough transform -----------")
 890            print("Central coordinate [ x, y ]: [{:.3f}, {:.3f}]".format(float(self.x), 
 891                                                                         float(self.y)))
 892            print("--------------------------------------------------------------------------")
 893        
 894        
 895        self.x, self.y, self.r = float(self.x[0]), float(self.y[0]), float(self.r[0])
 896
 897        # (7) Plot results (if required)
 898        if plot_results == 1:
 899           self.visualize_center(self.x, self.y, self.r) 
 900           
 901        # Return results, convert coordinates to float
 902        return self.x, self.y, self.r

1401            # a: the squared length of the side between the second 
1402            #    and third points (x[1], y[1]) and (x[2], y[2]).
1403            # b: the squared length of the side between the first 
1404            #    and third points (x[0], y[0]) and (x[2], y[2]).
1405            # c: the squared length of the side between the first 
1406            #    and second points (x[0], y[0]) and (x[1], y[1]).

1417        
1418        # radius
1419        ar = a**0.5
1420        br = b**0.5
1421        cr = c**0.5

1859    def ref_var(self, px, py, pr, plot_results = 0):
1860        '''         
1861        Adjust center coordinates of a detected circular diffraction pattern.
1862        The center adjustment is based on variance minimization.
1863        
1864        The 8-neighbourhood pixels (x) of the current center (o) 
1865        will be tested regarding the minimization:
1866    
1867        - x x x : (px - dx, py + dy) (px, py + dy) ( px + dx, py + dy)
1868    
1869        - x o x : (px - dx, py)      (px, py)      (px + dx, py)
1870    
1871        - x x x : (px - dx, py - dy) (px, py - dy) (px + dx, py - dy)
1872        
1873
1874        Parameters
1875        ----------
1876        self.image : array of uint8
1877            Input image in which the diffraction pattern is to be found
1878        px : float64
1879            x-coordinate of the detected center to be adjusted
1880        py : float64
1881            y-coordinate of the detected center to be adjusted
1882        pr : float64
1883            radius of the detected center
1884        plot_results : integer (default = 1)
1885            Plot Detected center. The default is 1.
1886        
1887        Returns
1888        -------
1889        px : array of int32
1890           corrected x-coordinates of pixels from circle border
1891        py : array of int32
1892            corrected y-coordinates of pixels from circle border
1893        pr : array of int32
1894            radius of the detected center
1895            
1896        '''
1897        
1898        # Store input for plot
1899        bckup = [np.copy(px), np.copy(py), np.copy(pr)]
1900        
1901        # Load image
1902        image = np.copy(self.parent.image)
1903
1904        # Starting values to be modified 
1905        init_var = self.intensity_var(image, px, py, pr)
1906        min_intensity_var = self.intensity_var(image, px, py, pr)
1907        best_center = (np.copy(px), np.copy(py))
1908        best_radius = np.copy(pr)
1909    
1910        # Convergence criterion for termination of gradient optimization 
1911        # (1) small positive value that serves as a threshold to determine 
1912        #     when the optimization process has converged
1913        convergence_threshold = 0.1*min_intensity_var
1914        
1915        # (2) maximum number of iterations of optimization
1916        max_iterations = 10
1917        
1918        # (3) keep track of the number of consecutive iterations where there 
1919        #     is no improvement in the objective function beyond 
1920        #     the convergence threshold
1921        no_improvement_count = 0
1922    
1923    
1924        # iterative refinement of the center of a circle while keeping
1925        # the radius constant.
1926        step = 0.3
1927        neighbors = [(float(dx), float(dy))
1928            for dx in np.arange(-1, 1 + step, step)
1929            for dy in np.arange(-1, 1 + step, step)]
1930        
1931        for iteration in range(max_iterations):    
1932            # Refine center while keeping radius constant
1933            curr = self.intensity_var(image, 
1934                                      best_center[0], 
1935                                      best_center[1], 
1936                                      best_radius) 
1937            # Store intensity sums of the current center's neighborhood
1938            curr_intensity_var = []
1939            for dx, dy in neighbors:
1940                nx, ny = best_center[0] + dx, best_center[1] + dy
1941                # Check if the point is within the expanded search radius
1942                curr_intensity_var.append(self.intensity_var(image, 
1943                                                             nx, ny, 
1944                                                             best_radius))
1945            
1946            # Find the minimum value coordinates within curr_intensity_var
1947            cx, _ = np.unravel_index(np.argmin(curr_intensity_var),
1948                                     [len(curr_intensity_var),1])
1949                    
1950            # Check for improvement of criterion -- in each iteration just once,
1951            # as the algorithm checks the neighbourhood of the best center (in
1952            # each iteration, the center is updated if possible)
1953            if min(curr_intensity_var) <= min_intensity_var:                           
1954                min_intensity_var = max(curr_intensity_var)
1955                
1956                # Calculate the new best coordinates of the center
1957                n = neighbors[cx]
1958                (nx, ny) = tuple(map(lambda x, y: float(x) + float(y), 
1959                                     best_center, n))

2306    '''
2307    Simple center determination for a symmetric diffractogram.
2308    
2309    * The center is determined as a center of intensity.
2310    * This works well for simple, symmetric diffraction patters, which are:
2311      (i) without beamstopper, (ii) pre-centered, and (iii) powder-like.
2312    * A real-life example of a simple symmetric diffractogram:
2313      a good powder electron diffraction pattern from STEMDIFF software.
2314    * This class is a legacy from previous EDIFF versions;
2315      it is kept mostly for backward compatibility.
2316      The functions in this class can be (and should be)
2317      replaced by a simple call of ediff.center.CenterLocator object.
2318      
2319    >>> # Center determination in a simple symmetric diffraction pattern
2320    >>> # (center = just center_of_intensity, no refinement
2321    >>>
2322    >>> # (1) Old way = this (old, legacy) IntensityCenter class:
2323    >>> xc,yc = ediff.center.IntensityCenter.center_of_intensity(
2324    >>>     arr, csquare=30, cintensity=0.8)
2325    >>>
2326    >>> # (2) New way = newer (and more universal) CenterLocator class:
2327    >>> xc,yc = ediff.center.CenterLocator(
2328    >>>     arr, detection_method='intensity', csquare=30, cintensity=0.8)
2329    '''
2330    
2331    @staticmethod
2332    def center_of_intensity(arr, csquare=20, cintensity=0.8):
2333        '''
2334        Find center of intensity/mass of an array.
2335        
2336        Parameters
2337        ----------
2338        arr : 2D-numpy array
2339            The array, whose intensity center will be determined.
2340        csquare : int, optional, default is 20
2341            The size/edge of the square in the (geometrical) center.
2342            The intensity center is searched only within the central square.
2343            Reasons: To avoid other spots/diffractions and
2344            to minimize the effect of an intensity assymetry around center. 
2345        cintensity : float, optional, default is 0.8
2346            The intensity fraction.
2347            When searching the intensity center, we will consider only
2348            pixels with intensity > max.intensity.
2349            
2350        Returns
2351        -------
2352        xc,yc : float,float
2353            XY-coordinates of the intensity/mass center of the array.
2354            Round XY-coordinates if you use them for image/array calculations.    
2355        '''
2356        # Get image/array size
2357        xsize,ysize = arr.shape
2358        # Calculate borders around the central square
2359        xborder = (xsize - csquare) // 2
2360        yborder = (ysize - csquare) // 2
2361        # Create central square = cut off the borders
2362        arr2 = arr[xborder:-xborder,yborder:-yborder].copy()
2363        # In the central square, set all values below cintenstity to zero
2364        arr2 = np.where(arr2>np.max(arr2)*cintensity, arr2, 0)
2365        # Calculate 1st central moments of the image
2366        M = sk.measure.moments(arr2,1)
2367        # Calculate the intensity center = centroid according to www-help
2368        (xc,yc) = (M[1,0]/M[0,0], M[0,1]/M[0,0])
2369        # We have centroid of the central square => recalculate to whole image
2370        (xc,yc) = (xc+xborder,yc+yborder)
2371        
2372        ## Return the final center
2373        return(xc,yc)

  • The center is determined as a center of intensity.
  • This works well for simple, symmetric diffraction patters, which are: (i) without beamstopper, (ii) pre-centered, and (iii) powder-like.
