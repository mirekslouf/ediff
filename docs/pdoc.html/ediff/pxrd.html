272    def gaussian(X,m,s):
273        '''
274        Gaussian function (~ profile for PXRD calculation).
275    
276        Parameters
277        ----------
278        X : numpy array
279            X-variable of Gaussian function.
280        m : float
281            Mean value = the center of Gaussian function.
282        s : float
283            Standard deviation = the width of Gaussian function.
284    
285        Returns
286        -------
287        NumPy array
288            The array with Y-values of the function; Y = Gaussian(X).
289        '''
290        return( 1/(s*np.sqrt(2*np.pi)) * np.exp(-(X-m)**2/(2*s**2)) )
291    
292    def lorentzian(X,m,s):
293        '''
294        Lorentzian function (~ profile for PXRD calculation).
295    
296        Parameters
297        ----------
298        X : numpy array
299            X-variable of Lorentzian function.
300        m : float
301            Mean value = the center of Lorentzian function.
302        s : float
303            Standard deviation = the width of Lorentzian function.
304    
305        Returns
306        -------
307        NumPy array
308            The array with Y-values of the functin; Y = Lorenzian(X).
309        '''
310        return( 1/np.pi * s/((X-m)**2 + s**2) )
311    
312    def pseudo_voigt(X,m,s,n=0.5):
313        '''
314        Pseudo-Voigt function (~ profile for PXRD calculation).
315    
316        Parameters
317        ----------
318        X : numpy array
319            X-variable of pseudo-Voigt function.
320        m : float
321            Mean value = the center of pseudo-Voigt function.
322        s : float
323            Standard deviation = the width of pseudo-Voigt function.
324        n : float, optional, the default is 0.5
325            Ratio between Gaussian and Lorentzian components.
326            In the very first approximation, we can set n = 0.5.
327            For more details, see *Notes* subsection below.
328            
329        Returns
330        -------
331        NumPy array
332            The array with Y-values of the functin; Y = Lorenzian(X).
333    
334        Notes
335        -----
336        * Pseudo-Voigt function
337          = linear combination of Gaussian and Lorentzian function.
338        * In the 1st approximation:
339            * 50/50 combination => n = 0.5
340            * sg = sigma_Gaussian
341              defined so that sg and s yielded the same FWHM
342        * more details in:
343            * https://en.wikipedia.org/wiki/Voigt_profile
344            * https://lmfit.github.io/lmfit-py/builtin_models.html
345        '''
346        # (1) Calculate sg
347        # (sg = sigma_Gaussian - see docstring above for more details
348        sg = s/np.sqrt(2*np.log(2))
349        # (2) Calculate Pseudo-Voigt function
350        # (Pseudo-Voigt = n * Gausian + (1-n) * Lorenzian)
351        pseudo_voigt = \
352            n * PeakProfiles.gaussian(X,m,sg) \
353            + (1-n) * PeakProfiles.lorentzian(X,m,s)
354        # (3) Return calculated function    
355        return(pseudo_voigt)

418        th1,th2 = th1,th2 = self.experiment.two_theta_range

433            index = int(round((two_theta - th1) * 100))
434            # insert the diffraction intensity at calculated position
435            df.Intensity[index] = intensity
436        # (3) Simulated diffraction profile
437        # = diffractogram with intensity profiles
438        # (3a) Is peak_profile_type defined?
439        # (Argument {peak_profile_type} has default value - pseudo_voight
440        # (Therefore, it should be defined unless user specifies None value

442            print('Error: peak_profile_type is not defined!')

416        # (1) Base diffraction profile
417        # = diffractogram with zero intensities
418        # = interval theta_min to theta_max with step=0.001, zero intensities
419        th1,th2 = th1,th2 = self.experiment.two_theta_range
420        numpoints = (th2-th1) * 100 + 1

425            columns=['TwoTheta','Intensity'])

427        # = diffractogram with line intensities

429        diffractions = np.round(
430            np.array(self.diffractions[['TwoTheta','Ihkl']]),2)
431        # (2b) insert line diffractions into zero profile from step (1)
432        for two_theta, intensity in diffractions:
433            # calculate index = position of the diffraction in profile array
434            index = int(round((two_theta - th1) * 100))
435            # insert the diffraction intensity at calculated position
436            df.Intensity[index] = intensity
437        # (3) Simulated diffraction profile
438        # = diffractogram with intensity profiles
439        # (3a) Is peak_profile_type defined?
440        # (Argument {peak_profile_type} has default value - pseudo_voight

445        # (3b) The peak_peak_profile_type is defined => calculate difractogram
446        # Precalculate peak profile
